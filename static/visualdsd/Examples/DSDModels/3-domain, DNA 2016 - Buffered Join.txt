def unbind = 0.1126

new ty  @ bind, unbind

new tw  @ bind, unbind

  (new ta @ bind, unbind
  new a

  new c
  new td @ bind, unbind
  new d
  new tB @ bind, unbind

  | M * [B]{tx^*}
  | M * [x]:[ty^ d]:[td^ y]{ta^*}
  | M * <d td^>
  | M * {td^*}[y]
  | M * {ty^*}[d]
  | N * <c tB^ B>))
